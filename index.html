📄 .github/workflows/build.yml 
1name: Build Android APK
2on:
3  push:
4    branches:
5      - main
6      - master
7  workflow_dispatch:
8
9jobs:
10  build:
11    runs-on: ubuntu-latest
12
13    steps:
14      - name: Checkout Source Code
15        uses: actions/checkout@v4
16
17      - name: Setup Node.js
18        uses: actions/setup-node@v4
19        with:
20          node-version: 20
21
22      - name: Install Dependencies + Capacitor
23        run: |
24          npm install
25          npm install @capacitor/cli @capacitor/core @capacitor/android --save
26
27      - name: Build React App
28        run: npm run build
29
30      - name: Setup Java JDK 17
31        uses: actions/setup-java@v4
32        with:
33          distribution: 'temurin'
34          java-version: '17'
35
36      - name: Setup Android SDK
37        uses: android-actions/setup-android@v3
38
39      - name: Sync Capacitor
40        run: npx cap sync android
41
42      - name: Fix Kotlin Duplicates
43        run: |
44          cat >> android/build.gradle << 'EOF'
45          subprojects {
46              configurations.all {
47                  resolutionStrategy {
48                      force "org.jetbrains.kotlin:kotlin-stdlib:1.9.0"
49                      force "org.jetbrains.kotlin:kotlin-stdlib-jdk7:1.9.0"
50                      force "org.jetbrains.kotlin:kotlin-stdlib-jdk8:1.9.0"
51                  }
52              }
53          }
54          EOF
55
56      - name: Build APK (Debug)
57        run: |
58          cd android
59          chmod +x gradlew
60          ./gradlew assembleDebug \
61            -Pandroid.useAndroidX=true \
62            -Pandroid.enableJetifier=true
63
64      - name: Upload APK
65        uses: actions/upload-artifact@v4
66        with:
67          name: exam-maker-apk
68          path: android/app/build/outputs/apk/debug/app-debug.apk

⚠️ قبل از push این workflow، باید لوکال انجام بدی:
